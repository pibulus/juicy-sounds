The Entire Integration (3 Lines!):

<!-- 1. Import Juicy Sounds -->
<script type="module">
  import JuicySounds from 'juicy-sounds';
  const sounds = new JuicySounds();
  
  // 2. That's it! Add sounds to any element:
  button.onclick = () => sounds.play('click');
  button.onmouseenter = () => sounds.play('hover');
</script>

Or Use Auto-Attach:

// Automatically attach sounds to elements
sounds.autoAttach({
  'click': 'button',
  'hover': 'button, a',
  'success': '.success-action'
});
