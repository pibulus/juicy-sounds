30-Second Setup

import JuicySounds from 'juicy-sounds';
const sounds = new JuicySounds();

button.onclick = () => sounds.play('click');
button.onmouseenter = () => sounds.play('hover');

  style: 'mechanical',
  volume: 0.3
});

typewriter.attach('input, textarea');
